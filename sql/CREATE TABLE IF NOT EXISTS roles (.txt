CREATE TABLE IF NOT EXISTS roles (
    id VARCHAR(50) PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    permissions JSON NOT NULL, 
    is_default BOOLEAN DEFAULT FALSE
);

CREATE TABLE IF NOT EXISTS users (
    id VARCHAR(50) PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    role_id VARCHAR(50),
    points INT DEFAULT 0,
    image VARCHAR(255),
    phone VARCHAR(50),
    phone_is_public BOOLEAN DEFAULT FALSE,
    notification_preferences JSON,
    FOREIGN KEY (role_id) REFERENCES roles(id)
);

CREATE TABLE IF NOT EXISTS settings (
    id INT PRIMARY KEY DEFAULT 1,
    site_name VARCHAR(255),
    site_icon VARCHAR(255),
    site_name_color VARCHAR(50),
    default_task_image VARCHAR(255),
    point_goal INT,
    enable_points BOOLEAN,
    min_task_points INT,
    max_task_points INT,
    menu_visibility JSON,
    smtp_config JSON,
    notification_role_defaults JSON,
    email_templates JSON
);

CREATE TABLE IF NOT EXISTS tasks (
    id VARCHAR(50) PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    description TEXT,
    image VARCHAR(255),
    task_date DATETIME,
    category VARCHAR(100),
    points INT DEFAULT 0,
    volunteers_needed INT DEFAULT 1,
    is_completed BOOLEAN DEFAULT FALSE,
    repeat_frequency INT,
    repeat_interval VARCHAR(20),
    is_template BOOLEAN DEFAULT FALSE,
    created_by VARCHAR(50),
    estimated_time INT,
    FOREIGN KEY (created_by) REFERENCES users(id) ON DELETE SET NULL
);

CREATE TABLE IF NOT EXISTS task_signups (
    task_id VARCHAR(50),
    user_id VARCHAR(50),
    signup_date DATETIME DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (task_id, user_id),
    FOREIGN KEY (task_id) REFERENCES tasks(id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS shifts (
    id VARCHAR(50) PRIMARY KEY,
    date DATE NOT NULL,
    start_time VARCHAR(10),
    end_time VARCHAR(10),
    title VARCHAR(255),
    description TEXT
);

CREATE TABLE IF NOT EXISTS shift_roles (
    id VARCHAR(50) PRIMARY KEY,
    shift_id VARCHAR(50) NOT NULL,
    user_id VARCHAR(50) NULL,
    role_name VARCHAR(100) NOT NULL,
    FOREIGN KEY (shift_id) REFERENCES shifts(id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE SET NULL
);

CREATE TABLE IF NOT EXISTS shift_trades (
    id VARCHAR(50) PRIMARY KEY,
    shift_role_id VARCHAR(50) NOT NULL,
    offering_user_id VARCHAR(50) NOT NULL,
    accepting_user_id VARCHAR(50) NULL,
    status ENUM('PENDING', 'COMPLETED', 'CANCELLED') DEFAULT 'PENDING',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (shift_role_id) REFERENCES shift_roles(id) ON DELETE CASCADE,
    FOREIGN KEY (offering_user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (accepting_user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS gallery_images (
    id VARCHAR(50) PRIMARY KEY,
    name VARCHAR(255),
    path VARCHAR(255) NOT NULL,
    upload_date DATETIME DEFAULT CURRENT_TIMESTAMP,
    size INT,
    width INT,
    height INT,
    uploaded_by VARCHAR(50),
    tags JSON,
    FOREIGN KEY (uploaded_by) REFERENCES users(id) ON DELETE SET NULL
);

CREATE TABLE IF NOT EXISTS categories (name VARCHAR(100) PRIMARY KEY);
